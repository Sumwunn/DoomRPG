// Common Imp
ACTOR CommonImpRPG : CommonImp Replaces CommonImp
{
	Tag "Imp"

	States
	{
	Spawn:
		TROO A 0 NoDelay ACS_NamedExecuteAlways("MonsterInit", 0)
		Goto Super::Spawn
	See:
		TROO A 0 ACS_NamedExecuteAlways("MonsterInit", 0)
		Goto Super::See
	}
}
